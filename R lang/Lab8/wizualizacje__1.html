  ggplot(data = data, mapping = mapping) + 
    geom_density(fill = "blue", ...)

my_scatter <- function(data, mapping, ...){
  ggplot(data = data, mapping = mapping) + 
    geom_point(color = "orange", ...)
}

  data = dat2,
  columns = c("var1", "var2", "var3"),

library(ggplot2)

ggplot(
  books_checked_out,
  aes(
    x = Time,
    y = Total,
    fill = factor(Genre, levels = c("Mystery", "Fiction", "Biography"))
  )
) +
  geom_area() +
  scale_fill_manual(
    values = c(
      "Mystery" = "cornflowerblue",
      "Fiction" = "green3",
      "Biography" = "salmon"
    ),
    breaks = c("Biography", "Fiction", "Mystery"),
    name = "Genre"
  ) +
  labs(
    title = "Books Checked Out",
    x = "Time",
    y = "Total"
  )
